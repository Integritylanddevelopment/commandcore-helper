# agents/localization_agent.py

"""
Agent Name: localization_agent
Role: translation
Version: 1.0.0
Description: Translates app or site content while preserving tone and cultural nuance.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.localization_tools import run_logic

# Agent Metadata
agent_name = "localization_agent"
agent_description = "Translates app or site content while preserving tone and cultural nuance."
agent_role = "translation"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "text": "Welcome to our app!",
    "target_language": "es"
}
# Output Schema
output_format = {
    "translated": "\u00a1Bienvenido a nuestra aplicaci\u00f3n!"
}

class LocalizationAgent(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "translate" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("localization_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"text": "Welcome to our app!", "target_language": "es"},
                "output": {"translated": "\u00a1Bienvenido a nuestra aplicaci\u00f3n!"}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(LocalizationAgent)

# Dev stub
if __name__ == "__main__":
    bot = inject(LocalizationAgent())
    result = bot.run({"text": "Welcome to our app!", "target_language": "es"})
    print(json.dumps(result, indent=2))
