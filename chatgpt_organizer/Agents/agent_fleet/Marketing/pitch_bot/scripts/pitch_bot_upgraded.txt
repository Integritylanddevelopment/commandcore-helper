# agents/pitch_bot.py

"""
Agent Name: pitch_bot
Role: fundraising
Version: 1.0.0
Description: Crafts investor pitch drafts, slides, and narratives based on funding stage.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.pitch_tools import run_logic

# Agent Metadata
agent_name = "pitch_bot"
agent_description = "Crafts investor pitch drafts, slides, and narratives based on funding stage."
agent_role = "fundraising"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "stage": "Seed",
    "focus": "vision + traction"
}
# Output Schema
output_format = {
    "pitch": "We're building the emotional OS for relationships..."
}

class PitchBot(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "pitch" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("pitch_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"stage": "Seed", "focus": "vision + traction"},
                "output": {"pitch": "We're building the emotional OS for relationships..."}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(PitchBot)

# Dev stub
if __name__ == "__main__":
    bot = inject(PitchBot())
    result = bot.run({"stage": "Seed", "focus": "vision + traction"})
    print(json.dumps(result, indent=2))
