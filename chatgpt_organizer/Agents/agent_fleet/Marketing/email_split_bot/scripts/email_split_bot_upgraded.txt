# agents/email_split_bot.py

"""
Agent Name: email_split_bot
Role: growth
Version: 1.0.0
Description: Creates A/B testing variations of email campaigns for conversion testing.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.email_splitter import run_logic

# Agent Metadata
agent_name = "email_split_bot"
agent_description = "Creates A/B testing variations of email campaigns for conversion testing."
agent_role = "growth"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "original_subject": "Start your journey today",
    "body": "Welcome to our platform!"
}
# Output Schema
output_format = {
    "variant_a": {
        "subject": "Begin your adventure now",
        "body": "Let's get started with something amazing."
    },
    "variant_b": {
        "subject": "Ready to begin?",
        "body": "Join us and explore new possibilities."
    }
}

class EmailSplitBot(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "email split" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("email_test_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"original_subject": "Start your journey today", "body": "Welcome to our platform!"},
                "output": {"variant_a": {"subject": "Begin your adventure now", "body": "Let's get started with something amazing."}, "variant_b": {"subject": "Ready to begin?", "body": "Join us and explore new possibilities."}}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(EmailSplitBot)

# Dev stub
if __name__ == "__main__":
    bot = inject(EmailSplitBot())
    result = bot.run({"original_subject": "Start your journey today", "body": "Welcome to our platform!"})
    print(json.dumps(result, indent=2))
