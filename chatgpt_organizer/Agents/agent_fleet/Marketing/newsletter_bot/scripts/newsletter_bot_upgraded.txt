# agents/newsletter_bot.py

"""
Agent Name: newsletter_bot
Role: content
Version: 1.0.0
Description: Composes recurring newsletter drafts based on company updates and blog content.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.newsletter_writer import run_logic

# Agent Metadata
agent_name = "newsletter_bot"
agent_description = "Composes recurring newsletter drafts based on company updates and blog content."
agent_role = "content"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "company_updates": [
        "New product launch",
        "New hire"
    ],
    "blog_posts": [
        "AI Trends 2025"
    ]
}
# Output Schema
output_format = {
    "newsletter": "Welcome to our June update! We\u2019ve launched a new product..."
}

class NewsletterBot(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "newsletter" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("newsletter_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"company_updates": ["New product launch", "New hire"], "blog_posts": ["AI Trends 2025"]},
                "output": {"newsletter": "Welcome to our June update! We\u2019ve launched a new product..."}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(NewsletterBot)

# Dev stub
if __name__ == "__main__":
    bot = inject(NewsletterBot())
    result = bot.run({"company_updates": ["New product launch", "New hire"], "blog_posts": ["AI Trends 2025"]})
    print(json.dumps(result, indent=2))
