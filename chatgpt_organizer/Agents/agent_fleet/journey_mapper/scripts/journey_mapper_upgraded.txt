# agents/journey_mapper.py

"""
Agent Name: journey_mapper
Role: product
Version: 1.0.0
Description: Visualizes and analyzes end-to-end user journeys from entry to retention.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.journey_tools import run_logic

# Agent Metadata
agent_name = "journey_mapper"
agent_description = "Visualizes and analyzes end-to-end user journeys from entry to retention."
agent_role = "product"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "entry_point": "Landing Page",
    "final_action": "Upgrade to Pro"
}
# Output Schema
output_format = {
    "journey_map": [
        "Landing Page",
        "Signup",
        "Onboarding",
        "First Use",
        "Upgrade to Pro"
    ]
}

class JourneyMapper(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "journey" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("journey_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"entry_point": "Landing Page", "final_action": "Upgrade to Pro"},
                "output": {"journey_map": ["Landing Page", "Signup", "Onboarding", "First Use", "Upgrade to Pro"]}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(JourneyMapper)

# Dev stub
if __name__ == "__main__":
    bot = inject(JourneyMapper())
    result = bot.run({"entry_point": "Landing Page", "final_action": "Upgrade to Pro"})
    print(json.dumps(result, indent=2))
