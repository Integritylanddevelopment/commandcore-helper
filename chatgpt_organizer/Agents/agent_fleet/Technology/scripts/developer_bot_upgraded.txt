# agents/developer_bot.py

"""
Agent Name: developer_bot
Role: engineering
Version: 1.0.0
Description: Writes or improves code snippets based on task descriptions or bug reports.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.code_writer import run_logic

# Agent Metadata
agent_name = "developer_bot"
agent_description = "Writes or improves code snippets based on task descriptions or bug reports."
agent_role = "engineering"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "task": "write a Python function to reverse a string"
}
# Output Schema
output_format = {
    "code": "def reverse_string(s): return s[::-1]"
}

class DeveloperBot(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "code" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("dev_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"task": "write a Python function to reverse a string"},
                "output": {"code": "def reverse_string(s): return s[::-1]"}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(DeveloperBot)

# Dev stub
if __name__ == "__main__":
    bot = inject(DeveloperBot())
    result = bot.run({"task": "write a Python function to reverse a string"})
    print(json.dumps(result, indent=2))
