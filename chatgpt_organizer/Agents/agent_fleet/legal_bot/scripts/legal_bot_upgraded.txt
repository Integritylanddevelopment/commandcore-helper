# agents/legal_bot.py

"""
Agent Name: legal_bot
Role: legal
Version: 1.0.0
Description: Drafts basic legal documents and explains compliance language in simple terms.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.legal_tools import run_logic

# Agent Metadata
agent_name = "legal_bot"
agent_description = "Drafts basic legal documents and explains compliance language in simple terms."
agent_role = "legal"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "document_type": "NDA",
    "parties": [
        "Alex",
        "CompanyX"
    ]
}
# Output Schema
output_format = {
    "doc": "This NDA is between Alex and CompanyX to protect confidential information."
}

class LegalBot(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "legal" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("legal_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"document_type": "NDA", "parties": ["Alex", "CompanyX"]},
                "output": {"doc": "This NDA is between Alex and CompanyX to protect confidential information."}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(LegalBot)

# Dev stub
if __name__ == "__main__":
    bot = inject(LegalBot())
    result = bot.run({"document_type": "NDA", "parties": ["Alex", "CompanyX"]})
    print(json.dumps(result, indent=2))
