# agents/launch_planner.py

"""
Agent Name: launch_planner
Role: operations
Version: 1.0.0
Description: Designs and schedules product launch activities by team and phase.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.launch_tools import run_logic

# Agent Metadata
agent_name = "launch_planner"
agent_description = "Designs and schedules product launch activities by team and phase."
agent_role = "operations"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "launch_date": "2025-06-01",
    "phases": [
        "Pre-Launch",
        "Launch Week",
        "Post-Launch"
    ]
}
# Output Schema
output_format = {
    "tasks": [
        "QA complete",
        "Press kit release",
        "Onboarding emails"
    ]
}

class LaunchPlanner(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "launch" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("launch_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"launch_date": "2025-06-01", "phases": ["Pre-Launch", "Launch Week", "Post-Launch"]},
                "output": {"tasks": ["QA complete", "Press kit release", "Onboarding emails"]}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(LaunchPlanner)

# Dev stub
if __name__ == "__main__":
    bot = inject(LaunchPlanner())
    result = bot.run({"launch_date": "2025-06-01", "phases": ["Pre-Launch", "Launch Week", "Post-Launch"]})
    print(json.dumps(result, indent=2))
