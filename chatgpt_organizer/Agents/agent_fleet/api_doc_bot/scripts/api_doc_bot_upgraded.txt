# agents/api_doc_bot.py

"""
Agent Name: api_doc_bot
Role: engineering
Version: 1.0.0
Description: Generates API documentation from endpoint specs and code.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.api_doc_generator import run_logic

# Agent Metadata
agent_name = "api_doc_bot"
agent_description = "Generates API documentation from endpoint specs and code."
agent_role = "engineering"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "endpoint": "/api/users",
    "method": "GET",
    "params": [
        "id"
    ],
    "description": "Fetch user by ID"
}
# Output Schema
output_format = {
    "doc": "GET /api/users?id={id} - Returns user profile"
}

class ApiDocBot(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "api doc" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("api_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"endpoint": "/api/users", "method": "GET", "params": ["id"], "description": "Fetch user by ID"},
                "output": {"doc": "GET /api/users?id={id} - Returns user profile"}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(ApiDocBot)

# Dev stub
if __name__ == "__main__":
    bot = inject(ApiDocBot())
    result = bot.run({"endpoint": "/api/users", "method": "GET", "params": ["id"], "description": "Fetch user by ID"})
    print(json.dumps(result, indent=2))
