# agents/team_meeting_bot.py

"""
Agent Name: team_meeting_bot
Role: admin
Version: 1.0.0
Description: Schedules team meetings and proposes agendas based on recent work.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.meeting_tools import run_logic

# Agent Metadata
agent_name = "team_meeting_bot"
agent_description = "Schedules team meetings and proposes agendas based on recent work."
agent_role = "admin"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "team": "Product",
    "recap_needed": true
}
# Output Schema
output_format = {
    "meeting_plan": "Agenda: Roadmap Review, AI Flow Update"
}

class TeamMeetingBot(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "team meeting" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("meeting_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"team": "Product", "recap_needed": true},
                "output": {"meeting_plan": "Agenda: Roadmap Review, AI Flow Update"}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(TeamMeetingBot)

# Dev stub
if __name__ == "__main__":
    bot = inject(TeamMeetingBot())
    result = bot.run({"team": "Product", "recap_needed": true})
    print(json.dumps(result, indent=2))
