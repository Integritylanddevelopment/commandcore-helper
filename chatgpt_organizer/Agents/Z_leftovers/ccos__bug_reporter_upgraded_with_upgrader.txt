# agents/bug_reporter.py

"""
Agent Name: bug_reporter
Role: qa
Version: 1.0.0
Description: Logs and categorizes bugs from logs or user complaints.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.bug_parser import run_logic

# Agent Metadata
agent_name = "bug_reporter"
agent_description = "Logs and categorizes bugs from logs or user complaints."
agent_role = "qa"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "log": "App crash on Stripe checkout with error 500"
}
# Output Schema
output_format = {
    "bug_type": "Checkout failure",
    "severity": "high"
}

class BugReporter(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "bug" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("bug_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"log": "App crash on Stripe checkout with error 500"},
                "output": {"bug_type": "Checkout failure", "severity": "high"}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(BugReporter)

# Dev stub
if __name__ == "__main__":
    bot = inject(BugReporter())
    result = bot.run({"log": "App crash on Stripe checkout with error 500"})
    print(json.dumps(result, indent=2))
