# agents/peer_feedback_bot.py

"""
Agent Name: peer_feedback_bot
Role: hr
Version: 1.0.0
Description: Formats peer feedback into anonymized summaries with strengths and growth areas.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.feedback_formatter import run_logic

# Agent Metadata
agent_name = "peer_feedback_bot"
agent_description = "Formats peer feedback into anonymized summaries with strengths and growth areas."
agent_role = "hr"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "raw_feedback": [
        "Very collaborative",
        "Needs better time tracking"
    ]
}
# Output Schema
output_format = {
    "summary": {
        "strengths": [
            "collaborative"
        ],
        "growth": [
            "time tracking"
        ]
    }
}

class PeerFeedbackBot(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "peer feedback" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("feedback_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"raw_feedback": ["Very collaborative", "Needs better time tracking"]},
                "output": {"summary": {"strengths": ["collaborative"], "growth": ["time tracking"]}}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(PeerFeedbackBot)

# Dev stub
if __name__ == "__main__":
    bot = inject(PeerFeedbackBot())
    result = bot.run({"raw_feedback": ["Very collaborative", "Needs better time tracking"]})
    print(json.dumps(result, indent=2))
