# agents/retention_bot.py

"""
Agent Name: retention_bot
Role: growth
Version: 1.0.0
Description: Analyzes user churn and recommends lifecycle messaging strategies.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.retention_engine import run_logic

# Agent Metadata
agent_name = "retention_bot"
agent_description = "Analyzes user churn and recommends lifecycle messaging strategies."
agent_role = "growth"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "user_behavior": [
        "no login in 7 days"
    ],
    "segment": "free users"
}
# Output Schema
output_format = {
    "recommendation": "Send reactivation email with 10% promo"
}

class RetentionBot(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "retention" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("retention_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"user_behavior": ["no login in 7 days"], "segment": "free users"},
                "output": {"recommendation": "Send reactivation email with 10% promo"}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(RetentionBot)

# Dev stub
if __name__ == "__main__":
    bot = inject(RetentionBot())
    result = bot.run({"user_behavior": ["no login in 7 days"], "segment": "free users"})
    print(json.dumps(result, indent=2))
