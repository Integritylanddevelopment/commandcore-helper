# agents/seo_optimizer.py

"""
Agent Name: seo_optimizer
Role: seo
Version: 1.0.0
Description: Improves page content and meta-data for search engine performance.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.seo_tools import run_logic

# Agent Metadata
agent_name = "seo_optimizer"
agent_description = "Improves page content and meta-data for search engine performance."
agent_role = "seo"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "page_title": "AI love coach",
    "description": "Learn how to improve your relationship with AI."
}
# Output Schema
output_format = {
    "optimized_title": "AI Relationship Coach \u2013 Build a Healthier Bond",
    "optimized_description": "Get daily relationship tips powered by AI guidance."
}

class SeoOptimizer(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "seo" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("seo_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"page_title": "AI love coach", "description": "Learn how to improve your relationship with AI."},
                "output": {"optimized_title": "AI Relationship Coach \u2013 Build a Healthier Bond", "optimized_description": "Get daily relationship tips powered by AI guidance."}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(SeoOptimizer)

# Dev stub
if __name__ == "__main__":
    bot = inject(SeoOptimizer())
    result = bot.run({"page_title": "AI love coach", "description": "Learn how to improve your relationship with AI."})
    print(json.dumps(result, indent=2))
