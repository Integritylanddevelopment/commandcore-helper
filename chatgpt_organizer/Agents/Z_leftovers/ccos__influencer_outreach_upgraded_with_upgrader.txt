# agents/influencer_outreach.py

"""
Agent Name: influencer_outreach
Role: marketing
Version: 1.0.0
Description: Drafts outreach messages for influencers and tracks engagement.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.influencer_engager import run_logic

# Agent Metadata
agent_name = "influencer_outreach"
agent_description = "Drafts outreach messages for influencers and tracks engagement."
agent_role = "marketing"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "platform": "Instagram",
    "audience_size": 50000,
    "niche": "wellness"
}
# Output Schema
output_format = {
    "outreach_message": "We love your content! Want to collaborate on our wellness launch?"
}

class InfluencerOutreach(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "influencer" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("influencer_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"platform": "Instagram", "audience_size": 50000, "niche": "wellness"},
                "output": {"outreach_message": "We love your content! Want to collaborate on our wellness launch?"}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(InfluencerOutreach)

# Dev stub
if __name__ == "__main__":
    bot = inject(InfluencerOutreach())
    result = bot.run({"platform": "Instagram", "audience_size": 50000, "niche": "wellness"})
    print(json.dumps(result, indent=2))
