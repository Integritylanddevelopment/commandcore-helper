# agents/eng_feedback_bot.py

"""
Agent Name: eng_feedback_bot
Role: engineering
Version: 1.0.0
Description: Collects and summarizes feedback from the engineering team for retrospectives or planning.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.eng_feedback_tools import run_logic

# Agent Metadata
agent_name = "eng_feedback_bot"
agent_description = "Collects and summarizes feedback from the engineering team for retrospectives or planning."
agent_role = "engineering"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "sprint": "Q2 Sprint 4",
    "feedback": [
        "Too many bugs in release",
        "CI pipeline was slow"
    ]
}
# Output Schema
output_format = {
    "summary": "Engineering team flagged bugs and CI pipeline speed as major blockers."
}

class EngFeedbackBot(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "eng feedback" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("eng_feedback_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"sprint": "Q2 Sprint 4", "feedback": ["Too many bugs in release", "CI pipeline was slow"]},
                "output": {"summary": "Engineering team flagged bugs and CI pipeline speed as major blockers."}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(EngFeedbackBot)

# Dev stub
if __name__ == "__main__":
    bot = inject(EngFeedbackBot())
    result = bot.run({"sprint": "Q2 Sprint 4", "feedback": ["Too many bugs in release", "CI pipeline was slow"]})
    print(json.dumps(result, indent=2))
