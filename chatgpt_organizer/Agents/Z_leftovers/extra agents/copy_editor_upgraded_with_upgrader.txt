# agents/copy_editor.py

"""
Agent Name: copy_editor
Role: writing
Version: 1.0.0
Description: Polishes and corrects grammar, tone, and style for clarity and impact.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.editing_tools import run_logic

# Agent Metadata
agent_name = "copy_editor"
agent_description = "Polishes and corrects grammar, tone, and style for clarity and impact."
agent_role = "writing"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "text": "our product are the best",
    "tone": "professional"
}
# Output Schema
output_format = {
    "edited_text": "Our products are the best."
}

class CopyEditor(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "edit" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("editing_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"text": "our product are the best", "tone": "professional"},
                "output": {"edited_text": "Our products are the best."}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(CopyEditor)

# Dev stub
if __name__ == "__main__":
    bot = inject(CopyEditor())
    result = bot.run({"text": "our product are the best", "tone": "professional"})
    print(json.dumps(result, indent=2))
