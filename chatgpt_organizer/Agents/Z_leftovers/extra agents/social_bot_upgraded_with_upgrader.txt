# agents/social_bot.py

"""
Agent Name: social_bot
Role: marketing
Version: 1.0.0
Description: Drafts social captions and posts for brand announcements, blog promos, and holidays.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.social_writer import run_logic

# Agent Metadata
agent_name = "social_bot"
agent_description = "Drafts social captions and posts for brand announcements, blog promos, and holidays."
agent_role = "marketing"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "topic": "Product Launch",
    "platform": "Twitter"
}
# Output Schema
output_format = {
    "post": "We just launched something huge. Try it now!"
}

class SocialBot(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "social" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("social_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"topic": "Product Launch", "platform": "Twitter"},
                "output": {"post": "We just launched something huge. Try it now!"}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(SocialBot)

# Dev stub
if __name__ == "__main__":
    bot = inject(SocialBot())
    result = bot.run({"topic": "Product Launch", "platform": "Twitter"})
    print(json.dumps(result, indent=2))
