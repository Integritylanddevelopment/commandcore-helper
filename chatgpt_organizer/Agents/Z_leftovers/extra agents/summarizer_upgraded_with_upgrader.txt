# agents/summarizer.py

"""
Agent Name: summarizer
Role: ops
Version: 1.0.0
Description: Condenses long documents, meetings, or threads into actionable summaries.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.summary_engine import run_logic

# Agent Metadata
agent_name = "summarizer"
agent_description = "Condenses long documents, meetings, or threads into actionable summaries."
agent_role = "ops"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "input_type": "meeting transcript",
    "content": "We agreed to delay launch and test onboarding first..."
}
# Output Schema
output_format = {
    "summary": "Delay launch; test onboarding flow."
}

class Summarizer(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "summarize" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("summary_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"input_type": "meeting transcript", "content": "We agreed to delay launch and test onboarding first..."},
                "output": {"summary": "Delay launch; test onboarding flow."}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(Summarizer)

# Dev stub
if __name__ == "__main__":
    bot = inject(Summarizer())
    result = bot.run({"input_type": "meeting transcript", "content": "We agreed to delay launch and test onboarding first..."})
    print(json.dumps(result, indent=2))
