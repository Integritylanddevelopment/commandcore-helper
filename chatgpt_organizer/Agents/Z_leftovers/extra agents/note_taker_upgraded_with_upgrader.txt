# agents/note_taker.py

"""
Agent Name: note_taker
Role: admin
Version: 1.0.0
Description: Takes meeting notes from transcripts and creates action items.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.note_engine import run_logic

# Agent Metadata
agent_name = "note_taker"
agent_description = "Takes meeting notes from transcripts and creates action items."
agent_role = "admin"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "transcript": "We need to launch by June. Sarah owns the Stripe setup."
}
# Output Schema
output_format = {
    "summary": "Launch by June",
    "action_items": [
        "Sarah: Stripe setup"
    ]
}

class NoteTaker(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "notes" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("notes_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"transcript": "We need to launch by June. Sarah owns the Stripe setup."},
                "output": {"summary": "Launch by June", "action_items": ["Sarah: Stripe setup"]}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(NoteTaker)

# Dev stub
if __name__ == "__main__":
    bot = inject(NoteTaker())
    result = bot.run({"transcript": "We need to launch by June. Sarah owns the Stripe setup."})
    print(json.dumps(result, indent=2))
