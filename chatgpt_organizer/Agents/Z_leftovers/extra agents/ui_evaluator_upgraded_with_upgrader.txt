# agents/ui_evaluator.py

"""
Agent Name: ui_evaluator
Role: product
Version: 1.0.0
Description: Evaluates UI screens for usability, accessibility, and emotional clarity.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.ui_scanner import run_logic

# Agent Metadata
agent_name = "ui_evaluator"
agent_description = "Evaluates UI screens for usability, accessibility, and emotional clarity."
agent_role = "product"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "screen": "login",
    "feedback_type": "accessibility"
}
# Output Schema
output_format = {
    "issues_found": [
        "Low contrast",
        "Missing ARIA labels"
    ]
}

class UiEvaluator(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "ui" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("ui_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"screen": "login", "feedback_type": "accessibility"},
                "output": {"issues_found": ["Low contrast", "Missing ARIA labels"]}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(UiEvaluator)

# Dev stub
if __name__ == "__main__":
    bot = inject(UiEvaluator())
    result = bot.run({"screen": "login", "feedback_type": "accessibility"})
    print(json.dumps(result, indent=2))
