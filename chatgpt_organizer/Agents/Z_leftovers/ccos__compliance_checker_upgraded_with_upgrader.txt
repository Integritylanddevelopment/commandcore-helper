# agents/compliance_checker.py

"""
Agent Name: compliance_checker
Role: legal
Version: 1.0.0
Description: Scans content or code for legal and policy compliance issues.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.compliance_engine import run_logic

# Agent Metadata
agent_name = "compliance_checker"
agent_description = "Scans content or code for legal and policy compliance issues."
agent_role = "legal"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "content": "Ad copy claiming instant weight loss"
}
# Output Schema
output_format = {
    "compliance_status": "Violation",
    "reason": "Health claim not supported by FDA"
}

class ComplianceChecker(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "compliance" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("compliance_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"content": "Ad copy claiming instant weight loss"},
                "output": {"compliance_status": "Violation", "reason": "Health claim not supported by FDA"}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(ComplianceChecker)

# Dev stub
if __name__ == "__main__":
    bot = inject(ComplianceChecker())
    result = bot.run({"content": "Ad copy claiming instant weight loss"})
    print(json.dumps(result, indent=2))
