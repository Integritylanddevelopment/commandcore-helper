# agents/keyword_miner.py

"""
Agent Name: keyword_miner
Role: seo
Version: 1.0.0
Description: Extracts trending or high-conversion keywords from a content theme.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.keyword_tools import run_logic

# Agent Metadata
agent_name = "keyword_miner"
agent_description = "Extracts trending or high-conversion keywords from a content theme."
agent_role = "seo"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "topic": "relationship health",
    "audience": "young couples"
}
# Output Schema
output_format = {
    "keywords": [
        "emotional safety",
        "couples therapy",
        "secure attachment"
    ]
}

class KeywordMiner(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "keywords" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("seo_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"topic": "relationship health", "audience": "young couples"},
                "output": {"keywords": ["emotional safety", "couples therapy", "secure attachment"]}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(KeywordMiner)

# Dev stub
if __name__ == "__main__":
    bot = inject(KeywordMiner())
    result = bot.run({"topic": "relationship health", "audience": "young couples"})
    print(json.dumps(result, indent=2))
