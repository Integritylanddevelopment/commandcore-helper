# agents/analytics_bot.py

"""
Agent Name: analytics_bot
Role: analytics
Version: 1.0.0
Description: Provides insights and KPIs based on recent app or site data.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.analytics_engine import run_logic

# Agent Metadata
agent_name = "analytics_bot"
agent_description = "Provides insights and KPIs based on recent app or site data."
agent_role = "analytics"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "metric": "daily_active_users",
    "range": "last_7_days"
}
# Output Schema
output_format = {
    "result": "Average DAUs: 1832"
}

class AnalyticsBot(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "analytics" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("analytics_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"metric": "daily_active_users", "range": "last_7_days"},
                "output": {"result": "Average DAUs: 1832"}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(AnalyticsBot)

# Dev stub
if __name__ == "__main__":
    bot = inject(AnalyticsBot())
    result = bot.run({"metric": "daily_active_users", "range": "last_7_days"})
    print(json.dumps(result, indent=2))
