# agents/hr_bot.py

"""
Agent Name: hr_bot
Role: hr
Version: 1.0.0
Description: Handles basic HR queries and policy lookups for employees.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.hr_helper import run_logic

# Agent Metadata
agent_name = "hr_bot"
agent_description = "Handles basic HR queries and policy lookups for employees."
agent_role = "hr"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "query": "What\u2019s our vacation policy?"
}
# Output Schema
output_format = {
    "response": "Employees receive 15 days of paid vacation annually."
}

class HrBot(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "hr" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("hr_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"query": "What\u2019s our vacation policy?"},
                "output": {"response": "Employees receive 15 days of paid vacation annually."}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(HrBot)

# Dev stub
if __name__ == "__main__":
    bot = inject(HrBot())
    result = bot.run({"query": "What\u2019s our vacation policy?"})
    print(json.dumps(result, indent=2))
