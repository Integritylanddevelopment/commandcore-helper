# agents/voice_consistency_bot.py

"""
Agent Name: voice_consistency_bot
Role: brand
Version: 1.0.0
Description: Ensures that all written content matches the brand's tone and voice guide.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.voice_checker import run_logic

# Agent Metadata
agent_name = "voice_consistency_bot"
agent_description = "Ensures that all written content matches the brand's tone and voice guide."
agent_role = "brand"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "content": "Hey! Wanna join?",
    "target_tone": "professional"
}
# Output Schema
output_format = {
    "recommendation": "Change to: Would you like to join us?"
}

class VoiceConsistencyBot(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "voice" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("voice_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"content": "Hey! Wanna join?", "target_tone": "professional"},
                "output": {"recommendation": "Change to: Would you like to join us?"}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(VoiceConsistencyBot)

# Dev stub
if __name__ == "__main__":
    bot = inject(VoiceConsistencyBot())
    result = bot.run({"content": "Hey! Wanna join?", "target_tone": "professional"})
    print(json.dumps(result, indent=2))
