# agents/qa_bot.py

"""
Agent Name: qa_bot
Role: qa
Version: 1.0.0
Description: Reviews features and user reports to generate test cases and track QA status.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.qa_tools import run_logic

# Agent Metadata
agent_name = "qa_bot"
agent_description = "Reviews features and user reports to generate test cases and track QA status."
agent_role = "qa"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "feature": "user login",
    "platform": "mobile"
}
# Output Schema
output_format = {
    "test_cases": [
        "Login with email",
        "Login with Google",
        "Wrong password error"
    ]
}

class QaBot(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "qa" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("qa_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"feature": "user login", "platform": "mobile"},
                "output": {"test_cases": ["Login with email", "Login with Google", "Wrong password error"]}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(QaBot)

# Dev stub
if __name__ == "__main__":
    bot = inject(QaBot())
    result = bot.run({"feature": "user login", "platform": "mobile"})
    print(json.dumps(result, indent=2))
