# agents/value_prop_bot.py

"""
Agent Name: value_prop_bot
Role: marketing
Version: 1.0.0
Description: Refines value propositions into clear, benefit-driven customer messaging.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.value_tools import run_logic

# Agent Metadata
agent_name = "value_prop_bot"
agent_description = "Refines value propositions into clear, benefit-driven customer messaging."
agent_role = "marketing"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "product": "TogetherWe",
    "feature": "AI-guided relationship therapy"
}
# Output Schema
output_format = {
    "value_prop": "Feel more connected in 10 minutes a day with AI guidance"
}

class ValuePropBot(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "value" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("value_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"product": "TogetherWe", "feature": "AI-guided relationship therapy"},
                "output": {"value_prop": "Feel more connected in 10 minutes a day with AI guidance"}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(ValuePropBot)

# Dev stub
if __name__ == "__main__":
    bot = inject(ValuePropBot())
    result = bot.run({"product": "TogetherWe", "feature": "AI-guided relationship therapy"})
    print(json.dumps(result, indent=2))
