# agents/persona_builder.py

"""
Agent Name: persona_builder
Role: marketing
Version: 1.0.0
Description: Constructs user personas from interview notes, analytics, and behavioral patterns.
"""

import json
from agents.agent_core import BaseAgent
from core.scheduler import schedule_task
from core.messaging import handle_message
from core.webhooks import register_webhook
from core.training import register_training_example
from core.upgrader import inject
from registry.agent_registry import register_agent
from utils.persona_tools import run_logic

# Agent Metadata
agent_name = "persona_builder"
agent_description = "Constructs user personas from interview notes, analytics, and behavioral patterns."
agent_role = "marketing"
agent_version = "1.0.0"

# Input Schema
expected_input = {
    "notes": "This user is a busy mom who values speed and trust.",
    "source": "interview"
}
# Output Schema
output_format = {
    "persona": {
        "name": "Busy Mom",
        "traits": [
            "speed",
            "trust"
        ]
    }
}

class PersonaBuilder(BaseAgent):
    def __init__(self):
        super().__init__(
            name=agent_name,
            description=agent_description,
            role=agent_role,
            version=agent_version,
            input_schema=expected_input,
            output_schema=output_format
        )

    def run(self, context):
        try:
            result = run_logic(context)
            self.log("Task completed.")
            return result
        except Exception as e:
            self.error(f"Execution failed: {str(e)}")
            return {"error": str(e)}            

    def auto_schedule(self):
        schedule_task(agent_name, self.run, day_of_week="fri", hour=7)

    def message_hooks(self, message_data):
        if "persona" in message_data.get("intent", ""):
            return self.run(message_data.get("context", {}))
        return {"status": "ignored"}

    def webhook_handler(self, payload):
        context = payload.get("persona_context", {})
        return self.run(context)

    def training_data(self):
        examples = [
            {
                "input": {"notes": "This user is a busy mom who values speed and trust.", "source": "interview"},
                "output": {"persona": {"name": "Busy Mom", "traits": ["speed", "trust"]}}
            }
        ]
        for example in examples:
            register_training_example(agent_name, example["input"], example["output"])

# Register
register_agent(PersonaBuilder)

# Dev stub
if __name__ == "__main__":
    bot = inject(PersonaBuilder())
    result = bot.run({"notes": "This user is a busy mom who values speed and trust.", "source": "interview"})
    print(json.dumps(result, indent=2))
